Add-on G — Desktop 통합

설치됨 3 modules ₩29,000/월 (Desktop 라이선스 포함)
대상: AXLE Desktop 앱(Electron)을 설치한 사용자. 전제: apps/desktop이 실행 중. 메인 윈도우는 axle.io 임베드. 차이: 다른 Pack과 달리 OS 권한이 필요한 기능(공인인증서, 마이크, 외부 포털 자동화)을 묶음.

① 모듈 메타데이터

pack-g/module.config.ts

const PACK_G_MODULES = [
  { id: "portal-automation", label: "포털 자동화", route: "/automation",   perm: "automation:*",
    requiresDesktop: true,
    integrations: ["hometax","minwon24","insurance","venturein","koita"],
    pbc: ["crawler"] },
  { id: "cert-management",   label: "공동인증서",  route: "/certs",        perm: "certs:*",
    requiresDesktop: true },
  { id: "voice-recording",   label: "녹취",        route: "/recording",    perm: "recording:*",
    requiresDesktop: true,
    deps: { soft: ["meetings"] } }   // 미팅(A) install 시 자동 첨부
];
